# Start job discovery curl -H "Authorization: Bearer <token>" \ -X POST https://your-worker.workers.dev/api/runs/discovery # Semantic job search curl -H "Authorization: Bearer <token>" \ "https://your-worker.workers.dev/api/agent/query?q=senior python developer remote" # Generate optimized cover letter curl -H "Authorization: Bearer <token>" \ -H "Content-Type: application/json" \ -X POST https://your-worker.workers.dev/api/cover-letter \ -d '{ "job_title": "Senior Software Engineer", "company_name": "TechCorp Inc", "job_description_text": "We are seeking a senior software engineer...", "candidate_career_summary": "Experienced full-stack developer with 5 years..." }' # Configure email insights curl -H "Authorization: Bearer <token>" \ -H "Content-Type: application/json" \ -X PUT https://your-worker.workers.dev/api/email/configs \ -d '{"id":"default","frequency_hours":24,"include_statistics":true}'
